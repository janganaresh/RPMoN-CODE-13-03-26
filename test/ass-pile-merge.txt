UPDATE piles p
JOIN assessment a
  ON p.`Area ID` = a.`Area ID`
 AND p.`Table ID` = a.`Table ID`
 AND p.`Pile No`  = a.`Pile No`
JOIN users u
  ON a.`Contractor` = u.`User ID`
SET
    p.`Assessment Date`       = a.`Task Date`,
    p.`Assessment Contractor` = u.`User Name`,
    p.`Assessment Case`       = a.`Assessment case`,
    p.`Assessment Status`     = 'Assessment Done'
WHERE a.`Assessment Status` = 'OE Approved';



UPDATE piles p
JOIN remedy r
  ON p.`Area ID` = r.`Area ID`
 AND p.`Table ID` = r.`Table ID`
 AND p.`Pile No`  = r.`Pile No`
SET
    p.`Remedy Date`       = r.`Task Date`,
    p.`Remedy Status`     = 'Remedy Done',
    p.`Remedy Contractor` = r.`Remedy Text`
WHERE r.`Picture Location` IS NOT NULL
  AND r.`Picture Location` <> '';



SELECT
    `Area ID`,
    `Table ID`,
    `Pile No`,
    `Assessment Date`,
    `Assessment Case`,
    `Assessment Status`,
    `Assessment Contractor`,
    `Remedy Date`,
    `Remedy Status`,
    `Remedy Contractor`
FROM piles
WHERE `Assessment Status` = 'Assessment Done'
  AND `Area ID` <> 'A001';

function loadImageAsDataURL(url) {
    return new Promise((resolve, reject) => {
        const img = new Image();
        img.crossOrigin = "Anonymous";
        img.onload = () => {
            const canvas = document.createElement("canvas");
            canvas.width = img.width;
            canvas.height = img.height;
            const ctx = canvas.getContext("2d");
            ctx.drawImage(img, 0, 0);

            // 🔹 Keep same width/height but reduce file MB by lowering JPEG quality
            const compressedDataUrl = canvas.toDataURL("image/jpeg", 0.4); 
            // 0.5 = 50% quality (adjust 0.4–0.7 range for balance)

            resolve(compressedDataUrl);
        };
        img.onerror = reject;
        img.src = url;
    });
}
