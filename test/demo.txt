<script>
    function drawLightHeader(doc) {
    const pageWidth = doc.internal.pageSize.getWidth();

    // Light background (soft gray / lite blue)
    doc.setFillColor(153, 204, 255);   // very light color
    doc.rect(0, 0, pageWidth, 16, "F");

    // Header text
    doc.setTextColor(30, 41, 59);
    doc.setFont("helvetica", "bold");
    doc.setFontSize(15);
    doc.text(
        "Concrete Process Documentation",
        pageWidth / 2,
        11,
        { align: "center" }
    );

    // Reset text color
    doc.setTextColor(0, 0, 0);
}
async function saveConcretePDF() {

    const { jsPDF } = window.jspdf;

    const doc = new jsPDF({
        orientation: "landscape",
        unit: "mm",
        format: "a4"
    });

    const table = document.getElementById("reportTable");
    const pageWidth = doc.internal.pageSize.getWidth();

    drawLightHeader(doc);
let y = 24;

    let setsOnPage = 0;

    for (let i = 1; i < table.rows.length; i++) {

        if (setsOnPage === 3) {
    doc.addPage();
    drawLightHeader(doc);
    y = 24;
    setsOnPage = 0;
}

        const row = table.rows[i];
        const tableId = row.cells[0].innerText.trim();
        const imgTags = row.cells[1].querySelectorAll("img");

        doc.setFont("helvetica", "bold");
        doc.setFontSize(14);
        doc.text(`Table ID : ${tableId}`, 15, y);
        y += 8;

        let x = 15;
        const imgW = 32;
        const imgH = 40;

        for (let img of imgTags) {
            const dataURL = await loadImageAsDataURL(img.src);

            if (x + imgW > pageWidth - 15) {
                x = 15;
                y += imgH + 10;
            }

            doc.addImage(dataURL, "JPEG", x, y, imgW, imgH);
            x += imgW + 7;
        }

        y += imgH + 10;
        setsOnPage++;
    }
    const pageHeight = doc.internal.pageSize.getHeight();


doc.setFont("helvetica", "italic");
doc.setFontSize(10);
doc.setTextColor(120, 120, 120);

doc.text(
    "INDISOLAR - Software Generated Report",
    pageWidth / 2,
    pageHeight - 15,
    { align: "center" }
);
    doc.save("Concrete_Documentation.pdf");
}

function loadImageAsDataURL(url) {
    return new Promise((resolve) => {
        const img = new Image();
        img.crossOrigin = "Anonymous";
        img.onload = () => {
            const canvas = document.createElement("canvas");
            canvas.width = img.width;
            canvas.height = img.height;
            canvas.getContext("2d").drawImage(img, 0, 0);
            resolve(canvas.toDataURL("image/jpeg", 0.1));
        };
        img.src = url;
    });
}
